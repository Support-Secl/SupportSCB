	-- EXEC [sp_Lams_Disbrusment_Report_Daily] '2026-02-10','','PODsum' 
	ELSE IF @optType = 'podSum'
	BEGIN
		SELECT 
			NULL AS podTotalNoOfLoan,
			NULL AS podTotalNoOfIpf,
			NULL AS podTotalFigureBdt1,
			NULL AS podTotalFigureBdt2
		--FROM LamsDisbrusmentFinalData where reportingDate BETWEEN @startDtOfRptMonth AND @reportingDate
	END

	-- EXEC [sp_Lams_Disbrusment_Report_Daily] '2026-02-10','','closing' 
	ELSE IF @optType = 'closing'
	BEGIN
		;WITH ClosingData AS
		(
			SELECT
				branchOrVC,
				ReceivingDate                 AS closingDate,
				FacilityType,
				FacilityNo                    AS facilityRefNo,
				LoanMaster,
				NewLoanAcNo                   AS loanAccountNo,
				RIGHT('0000000000'+ REPLACE(NewLoanAcNo,'-',''),11)         AS eSuiteAccountNo,
				TransactionAcNo               AS transactionAccountNo,
				TRY_CONVERT(DECIMAL(18,2), REPLACE(LimitInFigure, ',', '')) AS limitInWords,
				TRY_CONVERT(DECIMAL(18,2), REPLACE(Commission, ',', ''))    AS commission,
				CustomerName,
				Remarks,
				TRY_CONVERT(DECIMAL(18,2), REPLACE(closingBalance, ',', '')) AS closingBalance,
				NULL                          AS actualLoanLimit,
				NULL                          AS category,
				TRY_CONVERT(DECIMAL(18,2), REPLACE(OutstandingAmount, ',', '')) AS outstandingAmount,
				ReasonForClosing,
				SentBy                        AS salesChannel,
				NULL                          AS expiryDate
			FROM lamsClosingFinalData
			WHERE ReportingDate BETWEEN @startDtOfRptMonth AND @reportingDate
			  AND reportType IN ('closing','part')   --  FULL + PART
		)
		SELECT
			ROW_NUMBER() OVER (ORDER BY closingDate, branchOrVC) AS slNo,
			*
		FROM ClosingData;
	END


	-- EXEC [sp_Lams_Disbrusment_Report_Daily] '2026-02-10','','closingSum' 
	ELSE IF @optType = 'closingSum'
	BEGIN
		;WITH ClosingSum AS
		(
			SELECT
				FacilityType,
				closingBalance
			FROM lamsClosingFinalData
			WHERE ReportingDate BETWEEN @startDtOfRptMonth AND @reportingDate
			  AND reportType IN ('closing','part')
		)
		SELECT
			/* =COUNTA(A:A) */
			COUNT(*) AS totalVolumeTotalNoOfClosure,

			/* CB Loan Closing (Non-PART) */
			COUNT(*) 
			- COUNT(CASE WHEN FacilityType LIKE '%part%' THEN 1 END)
			  AS totalVolumeCbLoanClosing,

			/* Excel: no logic */
			NULL AS totalVolumeStaffLoanClosing,

			/* Part Settlement */
			COUNT(CASE 
					WHEN FacilityType LIKE '%part%' 
					THEN 1 
				  END) AS totalVolumePartSettlement,

			/* Total Figure (Million BDT) */
			COALESCE(SUM(closingBalance) / 1000000, 0)
				AS totalFigureBdtTotalNoOfClosure,

			/* Excel: no logic */
			NULL AS totalFigureBdtCbLoanClosing,
			NULL AS totalFigureBdtStaffLoanClosing,
			NULL AS totalFigureBdtPartSettlement
		FROM ClosingSum;
	END
